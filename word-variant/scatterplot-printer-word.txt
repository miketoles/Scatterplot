  <div class="container">
    <div class="header">
      <div class="header-title">🖨️ Scatterplot Printer — Word</div>
      <div class="header-subtitle">Select files, set date, generate instructions for Word documents</div>
    </div>

    <div class="section">
      <div class="checkbox-container">
        <input type="checkbox" id="testMode" checked onchange="app.saveSettings()">
        <label for="testMode">🧪 Test Mode (Don't save files)</label>
      </div>
    </div>

    <div class="section">
      <div class="section-title">
        📁 Files to Print
        <span class="count-badge" id="fileCount">0</span>
      </div>

      <button class="btn btn-blue" onclick="app.selectFiles()" style="width: 100%; margin-bottom: 12px;">
        ➕ Add Word Files
      </button>
      <input type="file" id="fileInput" style="display:none" multiple accept=".doc,.docx,application/msword,application/vnd.openxmlformats-officedocument.wordprocessingml.document">

      <div style="font-size: 12px; color: var(--text-dim); margin-bottom: 12px;">
        Note: Browser cannot access full network paths. Use Windows app for full path support.
      </div>

      <div class="file-list" id="fileList">
        <div class="empty-state">
          <div class="empty-icon">📄</div>
          <div style="font-size: 14px; font-weight: 500;">No files selected</div>
        </div>
      </div>
    </div>

    <div class="section">
      <div class="section-title">🖨️ Print</div>

      <div class="date-row">
        <input type="date" id="printDate" class="input date-input">
        <button class="btn btn-secondary btn-small" onclick="app.setToday()">Today</button>
        <button class="btn btn-secondary btn-small" onclick="app.setTomorrow()">Tomorrow</button>
      </div>

      <div class="alert">
        <strong>Note:</strong> Generates instruction file. For automation, use Windows desktop app.
      </div>

      <button id="printBtn" class="btn btn-blue" style="width: 100%; margin-bottom: 8px;" onclick="app.generateInstructions()" disabled>
        📥 Generate Instructions
      </button>

      <button id="copyBtn" class="btn btn-secondary" style="width: 100%;" onclick="app.copyToClipboard()" disabled>
        📋 Copy to Clipboard
      </button>

  <script>
    const app = {
      files: [],
      config: { testMode: true },

      init() {
        this.loadSettings();
        this.setTomorrow();
        document.getElementById('fileInput').addEventListener('change', (e) => this.handleFileSelect(e));
      },

      loadSettings() {
        const saved = localStorage.getItem('scatterplot_settings_v2_word');
        if (saved) {
          this.config = JSON.parse(saved);
          document.getElementById('testMode').checked = this.config.testMode;
        }
      },

      saveSettings() {
        this.config.testMode = document.getElementById('testMode').checked;
        localStorage.setItem('scatterplot_settings_v2_word', JSON.stringify(this.config));
      },

      selectFiles() {
        document.getElementById('fileInput').click();
      },

      handleFileSelect(e) {
        Array.from(e.target.files).forEach(file => {
          if (!this.files.some(f => f.name === file.name && f.size === file.size)) {
            const fullPath = file.path || file.webkitRelativePath || file.name;
            this.files.push({
              file: file,
              name: file.name,
              path: fullPath,
              size: file.size
            });
          }
        });
        this.renderFileList();
        e.target.value = '';
      },

      removeFile(index) {
        this.files.splice(index, 1);
        this.renderFileList();
      },

      clearFiles() {
        if (this.files.length > 0 && confirm('Remove all files?')) {
          this.files = [];
          this.renderFileList();
        }
      },

      renderFileList() {
        const list = document.getElementById('fileList');
        const count = document.getElementById('fileCount');
        const printBtn = document.getElementById('printBtn');
        const copyBtn = document.getElementById('copyBtn');

        count.textContent = this.files.length;

        if (this.files.length === 0) {
          list.innerHTML = '<div class="empty-state"><div class="empty-icon">📄</div><div style="font-size: 14px; font-weight: 500;">No files selected</div></div>';
          printBtn.disabled = true;
          copyBtn.disabled = true;
          return;
        }

        printBtn.disabled = false;
        copyBtn.disabled = false;

        list.innerHTML = this.files.map((item, index) => `
          <div class="file-item">
            <div class="file-info">
              <div class="file-name">📄 ${item.name}</div>
              <div class="file-path">${item.path}</div>
              <div class="file-size">${this.formatFileSize(item.size)}</div>
            </div>
            <div class="file-remove" onclick="app.removeFile(${index})">✕</div>
          </div>
        `).join('');
      },

      formatFileSize(bytes) {
        if (bytes < 1024) return bytes + ' B';
        if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
        return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
      },

      setToday() {
        document.getElementById('printDate').value = this.formatDate(new Date());
      },

      setTomorrow() {
        const tomorrow = new Date();
        tomorrow.setDate(tomorrow.getDate() + 1);
        document.getElementById('printDate').value = this.formatDate(tomorrow);
      },

      formatDate(date) {
        return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;
      },

      formatDateForWord(dateString) {
        const date = new Date(dateString);
        const months = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
        return `${months[date.getMonth()]} ${date.getDate()}, ${date.getFullYear()}`;
      },

      buildInstructionsText() {
        const printDate = document.getElementById('printDate').value;
        const testMode = document.getElementById('testMode').checked;
        const formattedDate = this.formatDateForWord(printDate);

        let output = `SCATTERPLOT PRINT INSTRUCTIONS\n`;
        output += `Generated: ${new Date().toLocaleString()}\n\n`;
        output += `===========================================\n\n`;
        output += `PRINT DATE: ${formattedDate}\n`;
        output += `TEST MODE: ${testMode ? 'YES (Do not save)' : 'NO (Save files)'}\n`;
        output += `FILES: ${this.files.length}\n\n`;
        output += `===========================================\n\n`;
        
        if (testMode) {
          output += `⚠️ TEST MODE\n`;
          output += `1. Open document in Word\n2. Go to Header (Insert → Header) or check header/footer\n3. Look for "Date:" or "Date" text\n4. Replace ONLY the date part (keep any underscores)\n5. If no underscores exist, add 3 before and 3 after\n6. If no colon after "Date", add one\n7. Example: "Date 1/1/2026" becomes "Date: ___${formattedDate}___"\n8. Example: "Date: ___December 19, 2026___" becomes "Date: ___${formattedDate}___"\n9. Print\n10. Close WITHOUT SAVING\n\n`;
        } else {
          output += `✅ LIVE MODE\n`;
          output += `1. Open document in Word\n2. Go to Header (Insert → Header) or check header/footer\n3. Look for "Date:" or "Date" text\n4. Replace ONLY the date part (keep any underscores)\n5. If no underscores exist, add 3 before and 3 after\n6. If no colon after "Date", add one\n7. Example: "Date: ___December 19, 2026___" becomes "Date: ___${formattedDate}___"\n8. Example: "Date: 1/1/2026" becomes "Date: ___${formattedDate}___"\n9. SAVE (Ctrl+S)\n10. Print\n11. Close\n\n`;
        }

        output += `FILES:\n\n`;
        this.files.forEach((item, index) => {
          output += `${index + 1}. ${item.name}\n   Path: ${item.path}\n   Update to: ${formattedDate}\n\n`;
        });

        output += `\nDate format: "${formattedDate}"\nLocation: Word document header (top of page) or header/footer; look for "Date:" or "Date" text\n⚠️ IMPORTANT: \n- Keep any existing underscores before/after the date\n- If no underscores exist, add 3 before and 3 after\n- If no colon after "Date", add one\nExamples:\n"Date: ___December 19, 2026___" → "Date: ___${formattedDate}___"\n"Date: 1/1/2026" → "Date: ___${formattedDate}___"\n"Date 1/1/2026" → "Date: ___${formattedDate}___"\n`;
        return output;
      },

      async copyToClipboard() {
        if (this.files.length === 0 || !document.getElementById('printDate').value) {
          alert('⚠️ Add files and select date first');
          return;
        }

        const output = this.buildInstructionsText();
        try {
          await navigator.clipboard.writeText(output);
          alert('✅ Copied to clipboard!');
        } catch {
          const textarea = document.createElement('textarea');
          textarea.value = output;
          textarea.style.position = 'fixed';
          textarea.style.opacity = '0';
          document.body.appendChild(textarea);
          textarea.select();
          document.execCommand('copy');
          document.body.removeChild(textarea);
          alert('✅ Copied to clipboard!');
        }
      },

      generateInstructions() {
        if (this.files.length === 0 || !document.getElementById('printDate').value) {
          alert('⚠️ Add files and select date first');
          return;
        }

        const output = this.buildInstructionsText();
        const printDate = document.getElementById('printDate').value;
        
        const blob = new Blob([output], { type: 'text/plain;charset=utf-8' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `scatterplot-instructions-word-${printDate}.txt`;
        document.body.appendChild(a);
        a.click();
        setTimeout(() => {
          document.body.removeChild(a);
          URL.revokeObjectURL(url);
        }, 100);

        alert(`✅ Downloaded!\n\nFiles: ${this.files.length}\nDate: ${this.formatDateForWord(printDate)}`);
      }
    };

    window.addEventListener('load', () => app.init());
  </script>
